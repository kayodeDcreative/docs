# Quick Start

Get up and running with the Gately SDK in minutes.

## 1. Add the SDK

Choose one of these methods to add the Gately SDK to your project:

### Script Tag (Easiest)

```html
<script src="https://cdn.usegately.com/gately.min.js"></script>
```

### ES Module Import

```javascript
import { GatelyBrowserClient } from 'https://cdn.usegately.com/gately-sdk.esm.min.js'
```

## 2. Initialize the Client

```javascript
// Get your project ID from the Gately dashboard
const gately = new GatelyBrowserClient('your-project-id')
```

## 3. Basic Authentication

### Login with Email/Password

```javascript
try {
  const { user, session } = await gately.login('user@example.com', 'password')
  console.log('Logged in:', user.email)
} catch (error) {
  console.error('Login failed:', error.message)
}
```

### Check Authentication Status

```javascript
if (gately.isAuthenticated()) {
  const user = gately.getUser()
  console.log('User is logged in:', user.email)
} else {
  console.log('User is not logged in')
}
```

### Logout

```javascript
await gately.logout()
console.log('User logged out')
```

## 4. React Integration

### Using Script Tag

```jsx
// After loading the script tag
function LoginForm() {
  const { user, login, logout, isAuthenticated } = window.gately.useGately('your-project-id')
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')

  const handleLogin = async (e) => {
    e.preventDefault()
    try {
      await login(email, password)
    } catch (error) {
      console.error('Login failed:', error)
    }
  }

  if (isAuthenticated) {
    return (
      <div>
        <h1>Welcome, {user?.email}!</h1>
        <button onClick={logout}>Logout</button>
      </div>
    )
  }

  return (
    <form onSubmit={handleLogin}>
      <input
        type="email"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
        placeholder="Email"
        required
      />
      <input
        type="password"
        value={password}
        onChange={(e) => setPassword(e.target.value)}
        placeholder="Password"
        required
      />
      <button type="submit">Login</button>
    </form>
  )
}
```

## 5. Advanced Features

### Magic Link Authentication

```javascript
// Send magic link
await gately.sendMagicLink('user@example.com')
console.log('Magic link sent!')
```

### SSO Authentication

```javascript
// Google SSO
await gately.loginWithSSO('google')

// GitHub SSO
await gately.loginWithSSO('github')
```

### Form Submissions

```javascript
import { GatelyFormsClient } from 'https://cdn.usegately.com/gately-sdk.esm.min.js'

const forms = new GatelyFormsClient('your-project-id')

// Submit a form
const submission = await forms.submit('form-id', {
  name: 'John Doe',
  email: 'john@example.com',
  message: 'Hello world!'
})
```

### Analytics

```javascript
import { GatelyAnalyticsClient } from 'https://cdn.usegately.com/gately-sdk.esm.min.js'

const analytics = new GatelyAnalyticsClient('your-project-id')

// Track an event
await analytics.track('page_view', {
  page: '/home',
  referrer: 'google.com'
})
```

## 6. Complete Example

Here's a complete working example:

```html
<!DOCTYPE html>
<html>
<head>
  <title>Gately SDK Demo</title>
</head>
<body>
  <div id="app">
    <h1>Gately SDK Demo</h1>
    <div id="auth-status">Loading...</div>
    <div id="login-form" style="display: none;">
      <input type="email" id="email" placeholder="Email" />
      <input type="password" id="password" placeholder="Password" />
      <button onclick="handleLogin()">Login</button>
    </div>
    <div id="user-info" style="display: none;">
      <p>Welcome, <span id="user-email"></span>!</p>
      <button onclick="handleLogout()">Logout</button>
    </div>
  </div>

  <script src="https://cdn.usegately.com/gately.min.js"></script>
  <script>
    const gately = new window.GatelyBrowserClient('your-project-id')
    
    // Check auth status on load
    updateAuthStatus()
    
    async function handleLogin() {
      const email = document.getElementById('email').value
      const password = document.getElementById('password').value
      
      try {
        await gately.login(email, password)
        updateAuthStatus()
      } catch (error) {
        alert('Login failed: ' + error.message)
      }
    }
    
    async function handleLogout() {
      await gately.logout()
      updateAuthStatus()
    }
    
    function updateAuthStatus() {
      const authStatus = document.getElementById('auth-status')
      const loginForm = document.getElementById('login-form')
      const userInfo = document.getElementById('user-info')
      const userEmail = document.getElementById('user-email')
      
      if (gately.isAuthenticated()) {
        const user = gately.getUser()
        authStatus.textContent = 'Authenticated'
        loginForm.style.display = 'none'
        userInfo.style.display = 'block'
        userEmail.textContent = user.email
      } else {
        authStatus.textContent = 'Not authenticated'
        loginForm.style.display = 'block'
        userInfo.style.display = 'none'
      }
    }
  </script>
</body>
</html>
```
